    try (Connection conn = DatabaseUtil.getConnection();
         Statement stmt = conn.createStatement()) {

            if (rs.next()) {
                System.out.println("Versi√≥n de MySQL: " + rs.getString(1));

        }
        
        // Contar tablas en la base de datos
        try (ResultSet rs = stmt.executeQuery("SHOW TABLES")) {

                System.out.println(" - " + rs.getString(1));

        }
    } catch (SQLException e) {
        System.err.println("Error al obtener informaci√≥n: " + e.getMessage());

    private static void mostrarLogs(List<Log> logs) {

            System.out.println("Fecha: " + log.getFecha().format(formatter));
            System.out.println("Detalles: " + log.getDetalles());
        }
    }
